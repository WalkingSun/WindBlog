# 获取调用信息
```
package main

import (
	"fmt"
	"reflect"
	"runtime"
)

// 要检查的函数
func myFunction() {
	// 获取调用者的信息
	pc, file, line, ok := runtime.Caller(1)
	if !ok {
		fmt.Println("无法获取调用者信息")
		return
	}

	// 获取函数名
	funcName := runtime.FuncForPC(pc).Name()

	fmt.Printf("函数 %s 在文件 %s 的第 %d 行被调用\n", funcName, file, line)
}

func main() {
	myFunction()
}

```
